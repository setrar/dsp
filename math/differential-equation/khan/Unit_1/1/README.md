\begin{gather}
  \text{ in a system where } y = f(x) \to 

  \begin{cases}
    y^{''} + 2 y^{'} = 3y \\
    \\
    f^{''}(x) + 2 f^{'}(x) = 3 f(x) \\
    \\
    \frac{\mathit{d^2y}}{\mathit{dx^2}} + 2 \frac{\mathit{dy}}{\mathit{dx}} = 3y \\
    \\ 
  \end{cases} \\

\end{gather}
